def plot_contours(cov, pos, n = 2, ax = None, **kwargs):
    if ax is None:
        ax = plt.subplot(111, aspect='equal')
    D, v = np.linalg.eig(cov)
    S = np.sqrt(D)
    
    for j in range(1, n+1):
        ell = Ellipse(xy=pos,
                  width=S[0]*j*2, height=S[1]*j*2,
                  angle=np.rad2deg(np.arccos(v[0, 0])),
                  color = 'black', **kwargs)
        ell.set_facecolor('none')
        ax.add_artist(ell)
